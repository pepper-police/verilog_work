// テスト用の簡単なMIPSアセンブリプログラム
// 1+2+...+10の計算

// 初期値設定
// $1: カウンタ (1から10まで)
// $2: 合計値
// $3: 10 (上限値)
// $4: 一時的な計算結果

// $1 = 1
20010001 // addi $1, $0, 1
// $2 = 0 (合計値の初期化)
20020000 // addi $2, $0, 0
// $3 = 10 (上限値)
2003000A // addi $3, $0, 10

// ループ開始
// $2 = $2 + $1
00411020 // add $2, $2, $1
// $1 = $1 + 1
20210001 // addi $1, $1, 1
// if $1 <= $3 goto ループ開始
0023082A // slt $1, $1, $3
14200001 // bne $1, $0, -3

// 終了（無限ループ）
08000005 // j 5
